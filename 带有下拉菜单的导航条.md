  <meta charset="UTF-8">

<script src="//cdn.bootcss.com/bootstrap/3.3.7/js/bootstrap.js"></script>
<style>

</style>
<body>
<!-- Nav tabs -->
<ul class="nav nav-tabs" role="tablist">
  <li class="active"><a href="#home" data-toggle="tab">Home</a></li>
  <li><a href="#profile" data-toggle="tab">Profile</a></li>
  <li><a href="#messages" data-toggle="tab">Messages</a></li>
  <li class="dropdown">
    <a class="dropdown-toggle" href="#" data-toggle="dropdown">Dropdown
      <span class="caret"></span>
    </a>
    <ul class="dropdown-menu">
      <li><a data-toggle="tab" href="#1">Action</a></li>
      <li><a data-toggle="tab" href="#2">Another action</a></li>
      <li><a data-toggle="tab" href="#3">Something else here</a></li>
      <li><a data-toggle="tab" href="#4">Separated link</a></li>
    </ul>
  </li>
</ul>
